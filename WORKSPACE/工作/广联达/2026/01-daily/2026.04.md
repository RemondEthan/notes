git log --no-merges --format="%aN" | sort -u | while read author; do
  echo "=== $author ==="
  git log --no-merges --author="$author" --pretty=tformat: --numstat | \
    awk '{ add += $1; del += $2 } END { printf "  新增: %d 行\n  删除: %d 行\n  净增: %d 行\n\n", add, del, add - del }'
done
