FILER_SERVER="http://seaweedfs-filer-client.seaweedfs.svc.cluster.local:8888"

# 输出 DIRS 变量
echo "DIRS: $DIRS"

# 如果当前目录不存在，设置昨天为TODAY变量
if ! echo "$DIRS" | grep -q "$TODAY"; then
    TODAY="$YESTERDAY"
fi

# 输出调试信息
echo "当前日期 (TODAY): $TODAY"
echo "昨天日期 (YESTERDAY): $YESTERDAY"

for dir_name in $DIRS; do
    # 删除非今天的日志目录
    if [[ "$dir_name" =~ ^[0-9]{4}-[0-9]{2}-[0-9]{2}$ && "$dir_name" < "$TODAY" ]]; then
        echo "DELETE $FILER_SERVER$DIR_PATH/$dir_name?recursive=true"
        curl -X DELETE "$FILER_SERVER$DIR_PATH/$dir_name?recursive=true"
    fi
done

* GC调优：

| `-XX:+UseG1GC`              | 启用 G1 垃圾收集器                 | G1（Garbage-First）是一种现代化的垃圾回收器，适合大堆（>4GB）、低延迟的应用。它能自动划分堆为多个区域（Region），并优先回收垃圾最多的区域，减少停顿时间。 |
| --------------------------- | ---------------------------------- | ------------------------------------------------------------ |
| `-Xmn4g`                    | 设置新生代大小为 4GB               | `-Xmn` 是设置 **年轻代（Young Generation）** 的总大小。年轻代包括 Eden 区和两个 Survivor 区（S0/S1）。设置为 4g 表示新生代占 4GB 内存。增大新生代可以减少 Young GC 的频率。 |
| `-XX:G1HeapRegionSize=4m`   | 设置 G1 的每个 Region 大小为 4MB   | G1 将堆划分为多个等大小的 Region。默认是自动决定（1MB ~ 32MB）。设置为 4m 表示每个 Region 是 4MB。这会影响“巨型对象（Humongous Object）”的判定（>2MB 的对象就可能被视为巨型对象）。 |
| `-XX:MaxMetaspaceSize=512m` | 设置 Metaspace 最大为 512MB        | Metaspace 存放类的元数据（如类名、方法、字段等）。默认无上限，可能耗尽系统内存。设置上限可防止内存溢出。 |
| `-XX:MetaspaceSize=256m`    | 设置 Metaspace 初始大小为 256MB    | 当 Metaspace 使用达到此值时，会触发 GC 来回收无用的类。设置合理值可减少早期 Metaspace GC 次数。 |
| `-Xlog:gc*:gc.log:time`     | 开启 GC 日志，输出到文件并带时间戳 | <ul><li>`gc*`：记录所有 GC 相关日志</li><li>`gc.log`：日志输出到当前目录的 `gc.log` 文件</li><li>`time`：每条日志前加上时间戳（如 `[2025-08-08T15:20:30.123]`）</li></ul>这是 JDK 9+ 的新日志系统语法，用于分析 GC 性能。 |



 推荐操作流程：

1. **先用 `jstat -gc` 查看老年代是否持续增长**
2. **再用 `jmap -histo:live` 看是否有可疑的大对象**
3. **如果怀疑堆外内存，使用 `jcmd VM.native_memory summary`**
4. **确认后可执行 `jmap -dump` 生成 heap dump 文件**
5. **使用 `kubectl cp` 把 dump 文件复制出来，用 Eclipse MAT 或 jVisualVM 分析**

```sql
select state,count(*) from t_ds_process_instance group by state;

select state,count(*) from t_ds_task_instance group by state;

select f_state, count(*) from gdmp_task_center.t_tc_task_instance group by f_state;
```

* 直接更新：

```sql
update t_ds_process_instance set state=6 where state=1;

update t_ds_task_instance set state=6 where state=1;

update gdmp_task_center.t_tc_task_instance set f_state=6 where f_state=1;
```
